$$
\int f(x)dx = F(x) + C
$$

$$
\int_{a}^{b} f(x)dx = F(b)-F(a)
$$

$$
\varphi (x) = \int_{a}^{x}f(x)dx
$$

$$
\varphi (x) = \int_{a}^{x}f(t)dt
$$
$$
\varphi (x){}' = f(x)
$$
$$
\varphi (x) + C = F(x)
$$
$$
\varphi (x) = 0
$$
$$
C = F(a)
$$
$$
\varphi (x) + F(a) = F(x)
$$
$$
\varphi (b) + F(a) = F(b)
$$
$$
\varphi (b) = F(b) - F(a)
$$

\varphi (x)
